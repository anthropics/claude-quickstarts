      - Command: `todo list`

    </feature_2>

    <feature_3>

      - Command: `todo complete 1`

    </feature_3>

    <feature_4>
      - Delete a todo by number
      - Command: `todo delete 1`
      - Remove from todos.json
    </feature_4>

    <feature_5>

      - Command: `todo --help`
      - Display usage information
    </feature_5>
